DHCP CONFIGURATION (SERVER EXCLUSIONS)
---------------------------------------------------------------------------
! 1. IT Support (User PC) - Standard
! Exclude Gateway & beberapa IP cadangan
ip dhcp excluded-address 192.168.8.1 192.168.8.5
ip dhcp pool L1_ITSUPPORT
 network 192.168.8.0 255.255.255.192
 default-router 192.168.8.1
 dns-server 8.8.8.8
 exit

! 2. Server & DC (15 SERVER HOST)
! Total range host: .65 - .94 (30 Host)
! Gateway: .65
! Server Statik: 15 Server (.66 s.d .80)

ip dhcp pool L1_SERVER
 network 192.168.8.64 255.255.255.224
 default-router 192.168.8.65

 exit

! 3. Cybersecurity (2 SERVER HOST + Staff)
! Total range host: .97 - .126 (30 Host)

! Server Statik: 2 Server (.98 s.d .99)
